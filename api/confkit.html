36    def validate(self) -> bool:
37        """Validate that the value matches the expected type."""
38        orig_bases: tuple[type, ...] | None = getattr(self.__class__, "__orig_bases__", None)
39
40        if not orig_bases:
41            msg = "No type information available for validation."
42            raise InvalidConverterError(msg)
43
44        # Extract type arguments from the generic base
45        for base in orig_bases:
46            if hasattr(base, "__args__"):
47                type_args = base.__args__
48                if type_args:
49                    for type_arg in type_args:
50                        if hasattr(type_arg, "__origin__"):
51                            # For parameterized generics, check against the origin type
52                            if isinstance(self.value, type_arg.__origin__):
53                                return True
54                        elif isinstance(self.value, (self.type, type_arg)):
55                            return True
56                    msg = f"Value {self.value} is not any of {type_args}."
57                    raise InvalidConverterError(msg)
58        msg = "This should not have raised. Report to the library maintainers with code: `DTBDT`"
59        raise TypeError(msg)

61    @staticmethod
62    def cast_optional(default: T | None | BaseDataType[T]) -> BaseDataType[T | None]:
63        """Convert the default value to an Optional data type."""
64        if default is None:
65            return cast("BaseDataType[T | None]", NoneType())
66        return Optional(BaseDataType.cast(default))
67
68    @staticmethod
69    def cast(default: T | BaseDataType[T]) -> BaseDataType[T]:
70        """Convert the default value to a BaseDataType."""
71        # We use Cast to shut up type checkers, as we know primitive types will be correct.
72        # If a custom type is passed, it should be a BaseDataType subclass, which already has the correct types.
73        match default:
74            case bool():
75                data_type = cast("BaseDataType[T]", Boolean(default))
76            case None:
77                data_type = cast("BaseDataType[T]", NoneType())
78            case int():
79                data_type = cast("BaseDataType[T]", Integer(default))
80            case float():
81                data_type = cast("BaseDataType[T]", Float(default))
82            case str():
83                data_type = cast("BaseDataType[T]", String(default))
84            case BaseDataType():
85                data_type = default
86            case _:
87                msg = (
88                    f"Unsupported default value type: {type(default).__name__}. "
89                    "Use a BaseDataType subclass for custom types."
90                )
91                raise InvalidDefaultError(msg)
92        return data_type

163    @classmethod
164    def _detect_parser(cls) -> None:
165        """Set the parser for descriptors based on the file extension of cls._file.
166
167        Uses msgspec-based parsers for yaml, json, toml. Defaults to dict structure.
168        Only sets the parser if there is no parser set.
169        """
170        if cls._file is UNSET:
171            msg = "Config file is not set. Use `set_file()`."
172            raise ValueError(msg)
173        match cls._file.suffix.lower():
174            case ".ini":
175                cls._parser = ConfigParser()
176            case ".yaml" | ".yml" | ".json" | ".toml":
177                from confkit.ext.parsers import MsgspecParser  # noqa: PLC0415  Only import if actually used.
178                cls._parser = MsgspecParser()
179            case _:
180                msg = f"Unsupported config file extension: {cls._file.suffix.lower()}"
181                raise ValueError(msg)

191    def validate_strict_type(self) -> None:
192        """Validate the type of the converter matches the desired type."""
193        if self._data_type.convert is UNSET:
194            msg = "Converter is not set."
195            raise InvalidConverterError(msg)
196
197        cls = self.__class__
198        self.__config_value = cls._parser.get(self._section, self._setting)
199        self.__converted_value = self.convert(self.__config_value)
200
201        if not cls.validate_types:
202            return
203
204        self.__converted_type = type(self.__converted_value)
205        default_value_type = type(self._data_type.default)
206
207        is_optional = self.optional or isinstance(self._data_type, Optional)
208        if (is_optional) and self.__converted_type in (default_value_type, NoneType):
209            # Allow None or the same type as the default value to be returned by the converter when _optional is True.
210            return
211        if self.__converted_type is not default_value_type:
212            msg = f"Converter does not return the same type as the default value <{default_value_type}> got <{self.__converted_type}>."  # noqa: E501
213            raise InvalidConverterError(msg)
214
215        # Set the data_type value. ensuring validation works as expected.
216        self._data_type.value = self.__converted_value
217        if not self._data_type.validate():
218            msg = f"Invalid value for {self._section}.{self._setting}: {self.__converted_value}"
219            raise InvalidConverterError(msg)

257    def __get__(self, obj: object, obj_type: object) -> VT:
258        """Get the value of the attribute."""
259        # obj_type is the class in which the variable is defined
260        # so it can be different than type of VT

334    @classmethod
335    def with_kwarg(
336        cls, section: str, setting: str, name: str | None = None, default: VT = UNSET,
337    ) -> Callable[[Callable[P, F]], Callable[P, F]]:
338        """Insert a config value into **kwargs to the wrapped method/function using this descriptor.
339
340        Use kwarg.get(`name`) to get the value.
341        `name` is the name the kwarg gets if passed, if None, it will be the same as `setting`.
342        Section parameter is just for finding the config value.
343        """
344        if name is None:
345            name = setting
346        if default is UNSET and not cls._parser.has_option(section, setting):
347            msg = f"Config value {section=} {setting=} is not set. and no default value is given."
348            raise ValueError(msg)
349
350        def wrapper(func: Callable[P, F]) -> Callable[P, F]:
351            @wraps(func)
352            def inner(*args: P.args, **kwargs: P.kwargs) -> F:
353                if default is not UNSET:
354                    cls._set_default(section, setting, default)
355                kwargs[name] = cls._parser.get(section, setting)
356                return func(*args, **kwargs)
357
358            return inner
359        return wrapper
